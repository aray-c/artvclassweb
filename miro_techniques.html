            document.getElementById('modalTitle').innerText = tech.title;

            document.getElementById('modalTitle').className = `text-4xl font-display font-bold mb-2 text-${tech.color === 'miroYellow' ? 'miroBlack' : tech.color}`;

            document.getElementById('modalLevel').innerText = tech.level;

            

            // Set Icon

            const iconContainer = document.getElementById('modalIcon');

            iconContainer.innerHTML = `<i data-lucide="${tech.icon}" class="w-full h-full"></i>`;

            iconContainer.className = `w-32 h-32 mb-6 text-${tech.color === 'miroYellow' ? 'miroBlack' : tech.color}`;


            // Tools List

            document.getElementById('modalTools').innerHTML = tech.tools.map(tool => `<li class="flex items-center gap-2"><span class="w-2 h-2 bg-miroBlack rounded-full"></span>${tool}</li>`).join('');


            // Steps

            document.getElementById('modalSteps').innerHTML = tech.steps.map(step => `<li>${step}</li>`).join('');


            // Tips & Example

            document.getElementById('modalTips').innerText = tech.tips;

            document.getElementById('modalExample').innerText = tech.example;


            modal.classList.remove('hidden');

            lucide.createIcons();

        }


        function closeModal() {

            modal.classList.add('hidden');

        }


        // Close on outside click

        modal.addEventListener('click', (e) => {

            if (e.target === modal) closeModal();

        });
